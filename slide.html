filter text to pure english string

1. set regex

	init(pattern: String, 
	     options: NSRegularExpression.Options = [])

pattern : The regular expression pattern to compile

let regexStr = "[^a-zA-Z]"
let regex = try? NSRegularExpression.init(pattern: regexStr, options: .caseInsensitive)

2. use regex

	func stringByReplacingMatches(in string: String, 
                      options: NSRegularExpression.MatchingOptions = [], 
                        range: NSRange, 
                 withTemplate templ: String) -> String

string : The string to search for values within.
options : The matching options to use.
range : The range of the string to search
template : The substitution template used when replacing matching instances

regex?.stringByReplacingMatches(in: self, 
                           options: [], 
                             range: NSRange(location: 0, length: self.count), 
                      withTemplate: "")
